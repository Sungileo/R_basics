weight <-c(45,55,65,75,86,74,67,98,50,66)
a <- character(length(weight))

for (i in 1:length(weight)) {
  if (weight[i] >= 90) {
    a[i] <- "A"
  } else if (weight[i] >= 80 & weight[i] < 90) {
    a[i] <- "B"
  } else if (weight[i] >= 70 & weight[i] < 80) {
    a[i] <- "C"
  } else if (weight[i] >= 60 & weight[i] < 70) {
    a[i] <- "D"
  } else {
    a[i] <- "F"
  }
}
