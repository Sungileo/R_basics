조건문 복습 (교재)

job.type <- 'A'

if(job.type=='B'){
  bonus <- 200
}else{
  bonus <- 100
}
print(bonus)
[1] 100

else 생략

job.type <- 'B'
bonus <- 100
if(job.type=='A'){
  bonus <- 200  #직무유형이 A일 때 실행.
}
print(bonus)
[1] 100

다중 if-else문

score <- 85

if(score>90){
  grade <- 'A'
}else if(score>80){
  grade <- 'B'
}else if(score>70){
  grade <- 'C'
}else if(score>60){
  grade <- 'D'
}else{
  grade <- 'F'
}
print(grade)
[1] "B"

조건문에서 논리 연산자의 사용

a <- 10

if(a>5&b>5){
  print(a+b)
}else{
  print("조건 미충족")
}
[1] "조건 미충족"
if(a>5|b>30){
  print(a*b)

  print("조건 미충족")
}
[1] 20
a <- 10
b <- 20

if(a > 5 & b > 5){
  c = a + b
}else if(a > 5 | b > 30){
  c = a * b
}
print(c)
[1] 30

ifelse문

a <- 10
b <- 20

if(a>b){
  c <- a
}else{
  c <- b
}
print(c)
[1] 20
a <- 10 
b <- 20
c <- ifelse(a>b, a, b) 
print(c)
[1] 20

iris에서 꽃잎의 길이에 따른 분류 작업

mylabel <- c()                          # 비어 있는 벡터 선언

for(i in 1:nrow(iris)){
  if(iris$Petal.Length[i] <= 1.6){      
    mylabel[i] <- 'L'                   # 꽃잎의 길이(Petal.Length)가 1.6 이하면 L
  }else if(iris$Petal.Length[i] >= 5.1){
    mylabel[i] <- 'H'                   # 꽃잎의 길이(Petal.Length)가 5.1 이상이면 H 
  }else{                                
    mylabel[i] <- 'M'                   # 나머지는 M
  }
}

print(mylabel)            #mylabel벡터 출력.

while문

  • break 반복문 실행중 조건 부합시 그만 실행하기

  • 1 ~ 10 까지 더하기 중 5 이상값 나오면 그만하기 (1+2+3+4+5)

sum <- 0

for(i in 1:10){
  sum <- sum+i
  if(i>=5){
    break
  }
}

sum
[1] 15

복습끝

  • next 조건 부합시 건너뛰기

  • i가 5,6,7,8이면 건너뛰어라 (파이썬의 continue함수와 같음)

sum <- 0
for(i in 1:10){
  sum <- sum+i
  if(i>=5 & i<=8){
    next
  }
  print(c(i,sum))
}
[1] 1 1
[1] 2 3
[1] 3 6
[1]  4 10
[1]  9 45
[1] 10 55
sum
[1] 55
  • 1~10까지 더하기 중 짝수는 건너뛰기(홀수만 더해라, 1+3+5+7+9)
sum <- 0
for(i in 1:10){
  if(i%%2==0){  
    next
  }
  sum=sum+i
}
sum
[1] 25
